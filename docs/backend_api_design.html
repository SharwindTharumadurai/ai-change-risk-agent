Key code — analyses router (core endpoint)
# routers/analyses.py from fastapi import APIRouter, UploadFile, File, Depends, HTTPException from app.agent.analyzer import run_analysis from app.middleware.auth import get_current_user from app.database import get_db from app.models.analysis import Analysis from app.schemas.analysis import AnalysisResponse import uuid, time router = APIRouter() @router.post("", response_model=AnalysisResponse) async def create_analysis( file: UploadFile = File(None), content: str = None, # pasted text alternative current_user = Depends(get_current_user), db = Depends(get_db) ): if not file and not content: raise HTTPException(400, "Provide a file or pasted content") start = time.time() raw = await file.read() if file else content.encode() file_text = raw.decode("utf-8", errors="ignore") # Run full AI analysis pipeline result = await run_analysis(file_text, file.filename if file else "pasted_code") # Persist to database analysis = Analysis( id=uuid.uuid4(), org_id=current_user.org_id, user_id=current_user.id, file_name=file.filename if file else "pasted_code", risk_score=result["risk_score"], verdict=result["verdict"], risk_level=result["risk_level"], status="complete", processing_ms=int((time.time()-start)*1000) ) db.add(analysis); db.commit(); db.refresh(analysis) return result
Key code — AI analyzer (agent orchestrator)
# agent/analyzer.py from openai import AsyncOpenAI from app.agent.prompts import SYSTEM_PROMPT, build_user_prompt from app.engines.risk_scorer import compute_score import json client = AsyncOpenAI() # reads OPENAI_API_KEY from env RESPONSE_SCHEMA = { "type": "json_schema", "json_schema": { "name": "risk_report", "strict": True, "schema": { "type": "object", "properties": { "risk_score": {"type": "integer"}, "risk_level": {"type": "string"}, "verdict": {"type": "string"}, "confidence": {"type": "number"}, "change_summary": {"type": "string"}, "findings": {"type": "array"}, "cost_impact": {"type": "object"}, "reasoning_summary": {"type": "string"} }, "required": ["risk_score","risk_level","verdict","findings"], "additionalProperties": False } } } async def run_analysis(file_content: str, filename: str) -> dict: response = await client.chat.completions.create( model="gpt-4o-mini", # swap to gpt-4o for demo response_format=RESPONSE_SCHEMA, messages=[ {"role": "system", "content": SYSTEM_PROMPT}, {"role": "user", "content": build_user_prompt(file_content, filename)} ], max_tokens=2000, temperature=0.1 # low temp = consistent, deterministic scoring ) return json.loads(response.choices[0].message.content)
Middleware stack
JWT authentication
Every protected route validates Bearer token. Extracts org_id and user_id for all DB queries. Returns 401 on missing/expired token.
CORS
Allows requests from Vercel frontend domain and localhost:3000 for development. Rejects all other origins.
Request logging
Structured JSON logs every request: method, path, status, duration_ms, user_id. Sent to Railway built-in log drain.
Error handling
Global exception handler returns consistent JSON: {"error": "message", "code": "ERROR_CODE"}. Never leaks stack traces to client.
