Detection logic — how the engine finds violations
CRITICAL Public S3 bucket detection
def check_s3_public_access(resource): # Check ACL if resource.get("acl") in ["public-read", "public-read-write"]: return finding(severity="CRITICAL", points=40, control="CIS-2.1.5") # Check block_public_access settings bpa = resource.get("block_public_acls", True) if not bpa: return finding(severity="CRITICAL", points=40, control="CIS-2.1.5")
CRITICAL IAM wildcard permissions detection
def check_iam_wildcards(policy_document): for statement in policy_document.get("Statement", []): actions = statement.get("Action", []) resources = statement.get("Resource", []) effect = statement.get("Effect", "") if effect == "Allow": if "*" in actions and "*" in resources: return finding(severity="CRITICAL", points=45) # full admin elif "*" in actions: return finding(severity="HIGH", points=25) # wildcard action only
HIGH Open security group detection
def check_open_ports(security_group): dangerous_ports = {22: "SSH", 3389: "RDP", 3306: "MySQL", 5432: "PostgreSQL"} for rule in security_group.get("ingress", []): cidrs = rule.get("cidr_blocks", []) port = rule.get("from_port") if "0.0.0.0/0" in cidrs or "::/0" in cidrs: if port in dangerous_ports: return finding(severity="CRITICAL", points=40, port=dangerous_ports[port]) elif port == -1: # all traffic return finding(severity="HIGH", points=30)
MEDIUM Encryption checks
def check_encryption(resource, resource_type): if resource_type == "aws_db_instance": if not resource.get("storage_encrypted", False): return finding(severity="CRITICAL", points=40, control="CIS-2.3.1") elif resource_type == "aws_ebs_volume": if not resource.get("encrypted", False): return finding(severity="MEDIUM", points=15, control="CIS-2.2.1")
