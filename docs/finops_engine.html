Cost calculation engine — core logic
# finops/calculator.py # Pricing lookup table (us-east-1, on-demand) EC2_PRICES = { "t3.micro": 0.0104, "t3.small": 0.0208, "t3.medium": 0.0416, "t3.large": 0.0832, "t3.xlarge": 0.1664, "m5.large": 0.096, } RDS_PRICES = { "db.t3.medium": 0.078, "db.t3.large": 0.156, "db.r5.large": 0.24, } HOURS_PER_MONTH = 730 def estimate_ec2_cost(instance_type: str, count: int = 1) -> float: hourly = EC2_PRICES.get(instance_type, 0) return hourly * HOURS_PER_MONTH * count def estimate_rds_cost(instance_class: str, multi_az: bool = False) -> float: hourly = RDS_PRICES.get(instance_class, 0) monthly = hourly * HOURS_PER_MONTH return monthly * 2 if multi_az else monthly def calculate_cost_delta(before: dict, after: dict) -> dict: # Compare resource configs before and after change cost_before = estimate_resource_cost(before) cost_after = estimate_resource_cost(after) delta = cost_after - cost_before return { "monthly_before_usd": cost_before, "monthly_after_usd": cost_after, "monthly_delta_usd": delta, "annual_delta_usd": delta * 12, "direction": "increase" if delta > 0 else "decrease" if delta < 0 else "no_change" }

Cost alert thresholds
🚨 High cost alert — delta > $500/mo
Flags change for mandatory FinOps review. Added to risk report as HIGH cost finding. Engineering manager notified. Contributes +15 points to risk score.
⚠ Medium cost alert — delta $100–$500/mo
Informational warning in risk report. Recommends FinOps team review before deployment. Contributes +5 points to risk score.
✓ Low cost impact — delta < $100/mo
Reported in cost breakdown section only. No score contribution. Informational for FinOps awareness.
